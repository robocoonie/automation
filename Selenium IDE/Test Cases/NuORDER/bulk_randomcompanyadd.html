Bulk_RandomCompanyAdd
getEval [storedVars['prod'] = prompt('Number of Companies to Add?')];
getEval ord = 0
stringConvert ${prod}
while ord < ${prod};
click //section[@id='application']/nav/ul/li[4]
waitForText //div[@id='alpha_list']/ol[2]/li[2] N
click //section[@id='contacts']/nav/ul/li
click //div[@id='add_company']/span
waitForElementPresent //section[@id='company_details']/div[6]/label
randomString 5,alphalower coname
randomString 3,alpha con2
randomString 10,numeric ph
randomString 7,alphalower web
randomString 5,numeric addnum
randomString 10,alpha addStreet
randomString 4,numeric addSuite
randomString 9,alpha addCity
type //section[@id='company_details']/div/input ${coname}tion ${con2}
type name=phone ${ph}
type name=website www.${web}.com
type name=line_1 ${addnum} ${addStreet} Blvd
type name=line_2 Suite ${addSuite}
type name=city ${addCity} Town
type name=state CA
type name=zip ${addnum}
type name=display_name ADR ${con2}
click //div[@id='form_save']/span
waitForElementPresent //div[@id='notification_notify']/div/span
click //div[@id='notification_notify']/div/div
getEval ord++;
endWhile
